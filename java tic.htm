JavaScript (script.js)

javascript
const board = document.getElementById('board');
const statusElement = document.getElementById('status');
let currentPlayer = 'X';
let boardState = ['', '', '', '', '', '', '', '', '']; // Empty board

const winningCombinations = [
  [0, 1, 2],
  [3, 4, 5],
  [6, 7, 8],
  [0, 3, 6],
  [1, 4, 7],
  [2, 5, 8],
  [0, 4, 8],
  [2, 4, 6]
];

// Create the Tic-Tac-Toe board
board.addEventListener('click', handleCellClick);

function handleCellClick(event) {
  const index = event.target.dataset.index;
  
  // Ignore if cell is already taken or if the game is over
  if (boardState[index] !== '' || statusElement.textContent !== '') return;

  // Mark the cell with current player's symbol
  boardState[index] = currentPlayer;
  event.target.textContent = currentPlayer;
  event.target.classList.add('taken');
  
  // Check for a winner
  if (checkWinner()) {
    statusElement.textContent = `${currentPlayer} Wins! Redirecting...`;
    setTimeout(() => redirectToNewPage(), 2000);
    return;
  }

  // Check for a draw
  if (boardState.every(cell => cell !== '')) {
    statusElement.textContent = 'It\'s a Draw! Redirecting...';
    setTimeout(() => redirectToNewPage(), 2000);
    return;
  }

  // Switch player
  currentPlayer = currentPlayer === 'X' ? 'O' : 'X';
  statusElement.textContent = `${currentPlayer}'s turn`;
}

function checkWinner() {
  return winningCombinations.some(combination => {
    const [a, b, c] = combination;
    return boardState[a] === currentPlayer && boardState[b] === currentPlayer && boardState[c] === currentPlayer;
  });
}

function redirectToNewPage() {
  window.location.href = "newpage.html"; // Change this to your desired URL
}

Explanation:

  1. HTML: We create the basic structure for the Tic-Tac-Toe game. It has a div to represent the game board, and each cell will be clickable. There is also a paragraph element to display the game status.

  2. CSS: The styles center the board on the screen, provide basic styles for the grid and the cells, and add interactivity with hover effects. The .taken class disables further clicks on cells once they are occupied.

  3. JavaScript:

    • boardState: An array that holds the state of each cell (empty, 'X', or 'O').
    • handleCellClick(): This function handles clicks on the cells, updates the board, and checks if there's a winner or a draw.
    • checkWinner(): This function checks whether a player has won by matching one of the predefined winning combinations.
    • redirectToNewPage(): This function redirects to the new page after a winner is found or after a draw.

Flow:

  • When a user clicks on a cell, it alternates between "X" and "O".
  • If either player wins, a message is displayed, and the page redirects after 2 seconds.
  • If all cells are filled without a winner, it shows a draw message and redirects after 2 seconds.

Customization:

  • Change "newpage.html" in the redirectToNewPage function to the URL of the page you want to redirect to after the game ends.

This setup gives a smooth transition from the game to a new page while ensuring that the user experience feels like a mini-game before the redirection.

4. Setting up HTTPS with Node.js (Using Express)

Now, let's serve these files over HTTPS. This involves setting up a simple HTTPS server using Node.js and Express. You'll need an SSL certificate to use HTTPS. If you're testing locally, you can generate a self-signed certificate.

Step 1: Install Dependencies

First, create a new Node.js project and install express and https.

bash
mkdir tictactoe-game
cd tictactoe-game
npm init -y
npm install express

Step 2: Generate a Self-Signed SSL Certificate (for local testing)

You can generate a self-signed certificate using openssl. In the terminal, run:

bash
openssl genpkey -algorithm RSA -out private.pem
openssl req -new -key private.pem -out cert.csr
openssl x509 -req -days 365 -in cert.csr -signkey private.pem -out cert.crt

This will generate two files: private.pem (your private key) and cert.crt (the certificate).

Step 3: Create an HTTPS Server (server.js)

Now, create a server.js file to serve your game files over HTTPS.

javascript
const express = require('express');
const https = require('https');
const fs = require('fs');
const path = require('path');

const app = express();

// Serve static files (HTML, CSS, JS)
app.use(express.static(path.join(__dirname, 'public')));

// Redirect to HTTPS if accessing HTTP
app.get('/', (req, res) => {
  res.sendFile(path.join(__dirname, 'public', 'index.html'));
});

// Start the HTTPS server
const options = {
  key: fs.readFileSync('private.pem'),
  cert: fs.readFileSync('cert.crt')
};

https.createServer(options, app).listen(3000, () => {
  console.log('HTTPS server running on https://localhost:3000');
});

Step 4: Organize the Project Structure

Organize your project directory to look like this:

  /public
    index.html
    styles.css
    script.js
  private.pem
  cert.crt
  server.js

Step 5: Run the Server

Finally, run the server:

bash
node server.js

Now, open your browser and go to https://localhost:3000. You should see your Tic-Tac-Toe game served over HTTPS.

Notes:

  • For production, use a trusted SSL certificate rather than a self-signed one.
  • If you're deploying to a cloud service (like AWS, Heroku, etc.), they often handle SSL certificates for you.

3. JavaScript (script.js)

javascript
const board = document.getElementById('board');
const statusElement = document.getElementById('status');
let currentPlayer = 'X';
let boardState = ['', '', '', '', '', '', '', '', '']; // Empty board

const winningCombinations = [
  [0, 1, 2],
  [3, 4, 5],
  [6, 7, 8],
  [0, 3, 6],
  [1, 4, 7],
  [2, 5, 8],
  [0, 4, 8],
  [2, 4, 6]
];

// Create the Tic-Tac-Toe board
board.addEventListener('click', handleCellClick);

function handleCellClick(event) {
  const index = event.target.dataset.index;
  
  // Ignore if cell is already taken or if the game is over
  if (boardState[index] !== '' || statusElement.textContent !== '') return;

  // Mark the cell with current player's symbol
  boardState[index] = currentPlayer;
  event.target.textContent = currentPlayer;
  event.target.classList.add('taken');
  
  // Check for a winner
  if (checkWinner()) {
    statusElement.textContent = `${currentPlayer} Wins! Redirecting...`;
    setTimeout(() => redirectToNewPage(), 2000);
    return;
  }

  // Check for a draw
  if (boardState.every(cell => cell !== '')) {
    statusElement.textContent = 'It\'s a Draw! Redirecting...';
    setTimeout(() => redirectToNewPage(), 2000);
    return;
  }

  // Switch player
  currentPlayer = currentPlayer === 'X' ? 'O' : 'X';
  statusElement.textContent = `${currentPlayer}'s turn`;
}

function checkWinner() {
  return winningCombinations.some(combination => {
    const [a, b, c] = combination;
    return boardState[a] === currentPlayer && boardState[b] === currentPlayer && boardState[c] === currentPlayer;
  });
}

function redirectToNewPage() {
  window.location.href = "newpage.html"; // Change this to your desired URL
}
